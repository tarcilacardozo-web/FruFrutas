    <script>
        // Função para alternar os campos de Entrega ou Retirada
        function alternarEntrega() {
            const tipo = document.querySelector('input[name="tipo_entrega"]:checked').value;
            const boxEntrega = document.getElementById('box-entrega');
            const boxRetirada = document.getElementById('box-retirada');

            if (tipo === 'entrega') {
                boxEntrega.classList.add('active');
                boxRetirada.classList.remove('active');
            } else {
                boxEntrega.classList.remove('active');
                boxRetirada.classList.add('active');
            }
        }

        // Função para formatar a mensagem e abrir o WhatsApp
        function enviarPedido() {
            const numeroWhatsApp = "5521999999999"; // COLOQUE SEU NÚMERO AQUI (DDD + NUMERO)
            
            const nome = document.getElementById('cli_nome').value;
            const pedido = document.getElementById('cli_pedido').value;
            const tipo = document.querySelector('input[name="tipo_entrega"]:checked').value;
            const pagamento = document.getElementById('cli_pagamento').value;
            
            if (!nome || !pedido) {
                alert("Por favor, preencha seu nome e o que deseja pedir!");
                return;
            }

            let textoMsg = `*NOVO PEDIDO - FRUFRUTAS* 🍉\n\n`;
            textoMsg += `👤 *Cliente:* ${nome}\n`;
            textoMsg += `📝 *Pedido:* ${pedido}\n\n`;
            
            if (tipo === 'entrega') {
                const endereco = document.getElementById('cli_endereco').value;
                if (!endereco) { alert("Preencha o endereço de entrega!"); return; }
                textoMsg += `🛵 *Modalidade:* Entrega\n`;
                textoMsg += `📍 *Endereço:* ${endereco}\n`;
                textoMsg += `*(Aguardando cálculo do frete)*\n\n`;
            } else {
                textoMsg += `🛍️ *Modalidade:* Retirada no Local\n\n`;
            }

            textoMsg += `💳 *Pagamento:* ${pagamento}`;

            // Codifica a mensagem e abre o app
            const url = `https://api.whatsapp.com/send?phone=${numeroWhatsApp}&text=${encodeURIComponent(textoMsg)}`;
            window.open(url, '_blank');
        }
    </script>
